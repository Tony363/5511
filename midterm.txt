Algo(A,n,v):
    i = 1
    j = 1
    k = n
    while j <= k
        if A[j] < v
            exchange A[i] with A[j]
            i = i + 1
            j = j + 1
        else if A[j] > v
            exchange A[j] with A[k]
            k = k - 1
        else
            j = j + 1
